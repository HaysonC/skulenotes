\section{Special Trigonometric Relationships}

\mybox{Sine ($\sin \theta$)}{
\[
\begin{array}{c|c}
\theta & \sin \theta \\
\hline
0^\circ & 0 \\
15^\circ & \frac{\sqrt{6} - \sqrt{2}}{4} \\
30^\circ & \frac{1}{2} \\
45^\circ & \frac{\sqrt{2}}{2} \\
60^\circ & \frac{\sqrt{3}}{2} \\
\end{array}
\]
}

\mybox{Cosine ($\cos \theta$)}{
\[
\begin{array}{c|c}
\theta & \cos \theta \\
\hline
0^\circ & 1 \\
15^\circ & \frac{\sqrt{6} + \sqrt{2}}{4} \\
30^\circ & \frac{\sqrt{3}}{2} \\
45^\circ & \frac{\sqrt{2}}{2} \\
60^\circ & \frac{1}{2} \\
\end{array}
\]
}

\mybox{Tangent ($\tan \theta$)}{
\[
\begin{array}{c|c}
\theta & \tan \theta \\
\hline
0^\circ & 0 \\
15^\circ & 2 - \sqrt{3} \\
30^\circ & \frac{1}{\sqrt{3}} \\
45^\circ & 1 \\
60^\circ & \sqrt{3} \\
\end{array}
\]
}


\mybox{Trigonometric Identities}{ 
\[
\begin{aligned}
\text{Pythagorean:} & \quad \sin^2 \theta + \cos^2 \theta = 1 \\
\text{also:} & \quad \sec^2\theta - 1 = \tan^2\theta \\
\text{Sine:} & \quad \sin(\theta + 180^\circ) = -\sin \theta \\
\text{also:} & \quad \sin(90^\circ - \theta) = \cos \theta \\
\text{Cosine:} & \quad \cos(\theta + 180^\circ) = -\cos \theta \\
\text{Tangent:} & \quad \tan(\theta + 180^\circ) = \tan \theta \\
\end{aligned}
\]
}

\end{multicols}
