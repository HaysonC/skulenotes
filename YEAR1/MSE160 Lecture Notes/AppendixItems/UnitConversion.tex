\section{Commonly Used Unit Conversions}
\begin{multicols}{2}

\mybox{Force}{
1 lbf (pound-force) = 4.44822 N \\
1 N (newton) = 0.224809 lbf \\
}

\mybox{Length}{
1 km = 1000 m \\
1 m = 100 cm = 1000 mm \\
1 inch = 2.54 cm \\
1 foot = 12 inches = 0.3048 m \\
1 yard = 3 feet = 0.9144 m \\
1 mile = 5280 feet = 1.60934 km \\
1 nm (nautical mile) = 1852 m \\
}

\mybox{Mass}{
1 ton (metric) = 1000 kg \\
1 kg = 1000 g \\
1 g = 1000 mg \\
1 lb (pound) = 0.453592 kg \\
1 oz (ounce) = 28.3495 g \\
}

\mybox{Speed}{
1 km/h = 0.621371 miles/hour \\
1 m/s = 3.6 km/h \\
}

\mybox{Time}{
1 hour = 60 minutes \\
1 minute = 60 seconds \\
1 day = 24 hours \\
1 year $\approx$ 365.25 days \\
}

\mybox{Temperature}{
$^\circ$F to $^\circ$C:
\begin{equation}
    T(^{\circ}\text{C}) = \frac{5}{9} (T(^{\circ}\text{F}) - 32)
\end{equation}
$^\circ$C to $^\circ$F:
\begin{equation}
    T(^{\circ}\text{F}) = \frac{9}{5} T(^{\circ}\text{C}) + 32
\end{equation}
K to $^\circ$C:
\begin{equation}
    T(^{\circ}\text{C}) = T(K) - 273.15
\end{equation}
$^\circ$C to K:
\begin{equation}
    T(K) = T(^{\circ}\text{C}) + 273.15
\end{equation}
}

\end{multicols}
