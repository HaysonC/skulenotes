        \( \sin(bt) \) & \( \frac{b}{s^2 + b^2} \) \\
        \( \cos(bt) \) & \( \frac{s}{s^2 + b^2} \) \\
        \( \sinh(bt) \) & \( \frac{b}{s^2 - b^2} \) (for \( s > |b| \)) \\
        \( \cosh(bt) \) & \( \frac{s}{s^2 - b^2} \) (for \( s > |b| \)) \\
        \( t^n e^{at} \) (for \( n = 0, 1, 2, \ldots \)) & \( \frac{n!}{(s - a)^{n+1}} \) (for \( s > a \)) \\
        \( e^{at} \sin(bt) \) & \( \frac{b}{(s - a)^2 + b^2} \) (for \( s > a \)) \\
        \( e^{at} \cos(bt) \) & \( \frac{s - a}{(s - a)^2 + b^2} \) (for \( s > a \)) \\
        \( t^n \) & \( \frac{n!}{s^{n+1}} \) \\
        \hline
